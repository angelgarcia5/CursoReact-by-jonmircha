1. useState ‚Üí Maneja el estado (datos que cambian)

Permite guardar valores y actualizar la interfaz cuando cambian.

function Contador() {
  const [contador, setContador] = useState(0); // valor inicial 0

  return (

      <h2>Contador: {contador}</h2>
      <button onClick={() => setContador(contador + 1)}>Sumar</button>
      <button onClick={() => setContador(0)}>Reiniciar</button>
    </div>
  );
}

3. useRef ‚Üí Accede directamente a elementos del DOM o guarda valores sin renderizar
import React, { useRef } from "react";

function EjemploUseRef() {
  const inputRef = useRef();

  const enfocar = () => {
    inputRef.current.focus();
  };

  return (
    <div>
      <input ref={inputRef} placeholder="Escribe algo..." />
      <button onClick={enfocar}>Enfocar input</button>
    </div>
  );
}

4. useContext ‚Üí Comparte datos entre componentes sin pasar props manualmente

Permite acceder a un contexto global, como un tema o usuario actual.

import React, { useContext, createContext } from "react";

const TemaContexto = createContext("claro");

function Hijo() {
  const tema = useContext(TemaContexto);
  return <p>El tema actual es: {tema}</p>;
}

function Padre() {
  return (
    <TemaContexto.Provider value="oscuro">
      <Hijo />
    </TemaContexto.Provider>
  );
}

5. useMemo y useCallback ‚Üí Optimizaci√≥n de rendimiento

Evitan c√°lculos o funciones innecesarias cada vez que se renderiza el componente.

import React, { useMemo, useState } from "react";

function EjemploUseMemo() {
  const [contador, setContador] = useState(0);

  const cuadrado = useMemo(() => {

    return contador * contador;
  }, [contador]);

  return (
    <div>
      <h2>Contador: {contador}</h2>
      <h3>Cuadrado: {cuadrado}</h3>
      <button onClick={() => setContador(contador + 1)}>Sumar</button>
    </div>
  );
}

6. useReducer ‚Üí Alternativa avanzada a useState

Se usa para manejar estados m√°s complejos (como Redux pero m√°s simple).

import React, { useReducer } from "react";

function reducer(state, action) {
  switch (action.type) {
    case "sumar":
      return { contador: state.contador + 1 };
    case "reset":
      return { contador: 0 };
    default:
      return state;
  }
}

function EjemploUseReducer() {
  const [state, dispatch] = useReducer(reducer, { contador: 0 });

  return (
    <div>
      <h2>Contador: {state.contador}</h2>
      <button onClick={() => dispatch({ type: "sumar" })}>Sumar</button>
      <button onClick={() => dispatch({ type: "reset" })}>Reset</button>
    </div>
  );
}
